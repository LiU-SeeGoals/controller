// Start a SSL Vision receiver, returns a channel from
// which SSL wrapper packets can be obtained.
func ( *GameState) ( string) {
	.ssl_receiver = receiver.NewSSLReceiver()
	.ssl_receiver.Connect()

	.ssl_receiver_channel = make(chan ssl_vision.SSL_WrapperPacket)
	go .ssl_receiver.Receive(.ssl_receiver_channel)
}
